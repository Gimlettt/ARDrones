319 public void CheckSecondQRCodePosition(Vector3 secondQRPosition, SecondQRCodeTransform qrTransform)
320 {
321 bool isVeryClose = false;
322 bool isKindOfNear = false;
323
324 if (currentCopterIndex > 0)
325 {
326 GameObject currentCylinder = props[currentCopterIndex - 1];
327 Vector3 localCylinderPosition = ARcamera.transform.InverseTransformPoint(currentCylinder.transform.position);
328 Vector3 localQRPosition = ARcamera.transform.InverseTransformPoint(secondQRPosition);
329 float distanceX = Mathf.Abs(localCylinderPosition.x - localQRPosition.x);
330 float distanceY = Mathf.Abs(localCylinderPosition.y - localQRPosition.y);
331
332 if (distanceX < 0.03f && distanceY < 0.03f)
333 {
334 isVeryClose = true;
335 Debug.Log("QR code is very close to the currently displaying cylinder.");
336 }
337 else if (distanceX < 0.03f && distanceY < 0.03f)
338 {
339 isKindOfNear = true;
340 Debug.Log("QR code is kind of near the currently displaying cylinder.");
341 }
342 }
343
344 if (qrTransform != null)
345 {
346 if (isVeryClose)
347 {
348 qrTransform.ChangeBallColor(Color.green);
349 }
350 else if (isKindOfNear)
351 {
352 qrTransform.ChangeBallColor(Color.green);
353 }
354 else
355 {
356 qrTransform.ChangeBallColor(Color.red);
357 }
358 }

368 private void UpdateArrowTransform(Vector3 copterPosition)
369 {
370 if (copterPosition.x <= boxPosition.x && copterPosition.y <= boxPosition.y)
371 {
372 arrow.transform.position = boxPosition + new Vector3(length_neg, height, -0.14f);
373 }
374 else if (copterPosition.x > boxPosition.x && copterPosition.y <= boxPosition.y)
375 {
376 arrow.transform.position = boxPosition + new Vector3(length_pos, height, -0.14f);
377 }
378 else if (copterPosition.x <= boxPosition.x && copterPosition.y > boxPosition.y)
379 {
380 arrow.transform.position = boxPosition + new Vector3(length_neg, -height, 0.15f);
381 }
382 else
383 {
384 arrow.transform.position = boxPosition + new Vector3(length_pos, -height, 0.15f);
385 }
386
387 Vector3 directionToCopter = copterPosition - arrow.transform.position;
388 directionToCopter.z = 0f;
389 float angle = Mathf.Atan2(directionToCopter.y, directionToCopter.x) * Mathf.Rad2Deg + 180f;
390 Quaternion targetRotation = Quaternion.Euler(0, 0, angle);
391 arrow.transform.rotation = targetRotation;
392 arrow.SetActive(true);
393 }
