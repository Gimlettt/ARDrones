95 void Update()
96 {
97 if (imageTargetBehaviour != null && !isFrozen)
98 {
99 TargetStatus targetStatus = imageTargetBehaviour.TargetStatus;
100 if (targetStatus.Status == Status.TRACKED)
101 {
102 Vector3 qrPosition = imageTargetBehaviour.transform.position;
103 Quaternion qrRotation = imageTargetBehaviour.transform.rotation;
104 Vector3 cameraToQR = qrPosition - arCamera.transform.position;
105 cameraToQR.Normalize();
106 Vector3 adjustedPosition = qrPosition + new Vector3(0, -0.045f, 0) + cameraToQR * 0.03f;
107 Box.transform.position = adjustedPosition;
108 Box.transform.rotation = qrRotation * Quaternion.Euler(-90, 0, 0);
109 Box.SetActive(true);
110 ColorChange(isFrozen);
111 propellerManager.UpdateCylinderPositions(adjustedPosition, qrRotation);
112 if (propellerManager.isDisplaying)
113 {
114 propellerManager.ChangePlugColor();
115 }
116 if (!hasWindowTrackedOnce)
117 {
118 lastKnownWindowPosition = qrPosition;
119 lastKnownWindowRotation = Quaternion.LookRotation(arCamera.transform.position - lastKnownWindowPosition);
120 UpdateWindowPosition();
121 hasWindowTrackedOnce = true;
122 }
123 }
124 else
125 {
126 Box.SetActive(false);
127 }
128 }
129 if (hasWindowTrackedOnce)

131 UpdateWindowPosition();
