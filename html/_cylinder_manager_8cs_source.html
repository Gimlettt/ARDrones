214 public void UpdateCylinderPositions(Vector3 qrPosition, Quaternion qrRotation)
215 {
216 boxPosition = qrPosition;
217 int configIndex = GetConfigurationIndex();
218
219 for (int i = 0; i < count; i++)
220 {
221 if (isDisplaying && i <= currentCopterIndex - 1)
222 {
223 // Update position for the currently displaying propeller
224 cylinders[i].transform.position = boxPosition + qrRotation * relativePositions[configIndex][i];
225 cylinders[i].transform.rotation = qrRotation * Quaternion.Euler(relativeRotations[configIndex][i]);
226 cylinders[i].SetActive(true);
227 UpdateArrowTransform(cylinders[i].transform.position);
228 }
229 else
230 {
231 // Update position for inactive propellers
232 cylinders[i].transform.position = boxPosition + qrRotation * relativePositions[configIndex][i];
233 cylinders[i].transform.rotation = qrRotation * Quaternion.Euler(relativeRotations[configIndex][i]);
234 cylinders[i].SetActive(false);
235 }
236 }

331 public void CheckSecondQRCodePosition(Vector3 secondQRPosition, SecondQRCodeTransform qrTransform)
332 {
333 bool isVeryClose = false;
334 bool isKindOfNear = false;
335
336 if (currentCopterIndex > 0)
337 {
338 GameObject currentCylinder = cylinders[currentCopterIndex - 1];
339 Vector3 localCylinderPosition = ARcamera.transform.InverseTransformPoint(currentCylinder.transform.position);
340 Vector3 localQRPosition = ARcamera.transform.InverseTransformPoint(secondQRPosition);
341
342 float distanceX = Mathf.Abs(localCylinderPosition.x - localQRPosition.x);
343 float distanceY = Mathf.Abs(localCylinderPosition.y - localQRPosition.y);
344
345 if (distanceX < 0.03f && distanceY < 0.03f)
346 {
347 isVeryClose = true;
348 Debug.Log("QR code is very close to the currently displaying cylinder.");
349 }
350 else if (distanceX < 0.03f && distanceY < 0.03f)
351 {
352 isKindOfNear = true;
353 Debug.Log("QR code is kind of near the currently displaying cylinder.");
354 }
355 }
356
357 if (qrTransform != null)
358 {
359 if (isVeryClose)
360 {
361 qrTransform.ChangeBallColor(Color.green);
362 }
363 else if (isKindOfNear)
364 {
365 qrTransform.ChangeBallColor(Color.green);
366 }
367 else
368 {
369 qrTransform.ChangeBallColor(Color.red);
370 }
371 }

381 private void UpdateArrowTransform(Vector3 copterPosition)
382 {
383 if (copterPosition.x <= boxPosition.x && copterPosition.y <= boxPosition.y)
384 {
385 arrow.transform.position = boxPosition + new Vector3(length_neg, height, -0.14f);
386 }
387 else if (copterPosition.x > boxPosition.x && copterPosition.y <= boxPosition.y)
388 {
389 arrow.transform.position = boxPosition + new Vector3(length_pos, height, -0.14f);
390 }
391 else if (copterPosition.x <= boxPosition.x && copterPosition.y > boxPosition.y)
392 {
393 arrow.transform.position = boxPosition + new Vector3(length_neg, -height, 0.15f);
394 }
395 else
396 {
397 arrow.transform.position = boxPosition + new Vector3(length_pos, -height, 0.15f);
398 }
399
400 Vector3 directionToCopter = copterPosition - arrow.transform.position;
401 directionToCopter.z = 0f;
402 float angle = Mathf.Atan2(directionToCopter.y, directionToCopter.x) * Mathf.Rad2Deg + 180f;
403 Quaternion targetRotation = Quaternion.Euler(0, 0, angle);
404 arrow.transform.rotation = targetRotation;
405 arrow.SetActive(true);
